160 [[nodiscard]] static bool containsCoupling(const MappedCouplings& couplings, const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
161 if (couplings.find(portTo) == couplings.end()) {
162 return false;
163 }
164 const auto& portsFrom = couplings.at(portTo);
165 return std::find(portsFrom.begin(), portsFrom.end(), portFrom) != portsFrom.end();

175 static void addCouplingToMap(MappedCouplings& coupList, const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
176 auto aux = coupList.find(portTo);
177 if (aux == coupList.end()) {
178 coupList[portTo] = {portFrom};
179 } else {
180 auto& portsFrom = aux->second;
181 if (std::find(portsFrom.begin(), portsFrom.end(), portFrom) != portsFrom.end()) {
182 throw CadmiumModelException("duplicate coupling");
183 }
184 portsFrom.push_back(portFrom);
185 }
186 }

194 void addCoupling(const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
195 if (!portTo->compatible(portFrom)) {
196 throw CadmiumModelException("invalid port type");
197 }
198 if (portFrom->getParent() == nullptr || portTo->getParent() == nullptr) {
199 throw CadmiumModelException("port does not belong to any model");
200 }
201 auto portFromParent = portFrom->getParent();
202 auto portToParent = portTo->getParent();
203 if (containsInPort(portFrom)) {
204 if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
205 addEIC(portFrom, portTo);
206 } else {
207 throw CadmiumModelException("invalid destination port");
208 }
209 } else if (portFromParent->getParent() == this && portFromParent->containsOutPort(portFrom)) {
210 if (containsOutPort(portTo)) {
211 addEOC(portFrom, portTo);
212 } else if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
213 addIC(portFrom, portTo);
214 } else {
215 throw CadmiumModelException("invalid destination port");
216 }
217 } else {
218 throw CadmiumModelException("invalid origin port");
219 }
220 }

326 void flatten(Coupled * parentPointer) {
327 // First, we identify the coupled subcomponents that need to be flattened
328 std::vector<std::shared_ptr<Coupled>> toFlatten;
329 for (auto& component: components) {
330 auto coupled = std::dynamic_pointer_cast<Coupled>(component.second);
331 if (coupled != nullptr) {
332 toFlatten.push_back(coupled);
333 }
334 }
335 // Next, we flatten all the coupled subcomponents
336 for (auto& coupled: toFlatten) {
337 coupled->flatten(this);
338 removeFlattenedCouplings(coupled);
339 components.erase(coupled->getId());
340 }
341 // Finally, we deserialize the resulting couplings. The model is already flat!
342 EIC = deserializeCouplings(serialEIC);
343 IC = deserializeCouplings(serialIC);
344 EOC = deserializeCouplings(serialEOC);
345
346 // If pointer to parent is not null, we propagate the flattening to the corresponding parent coupled model.
347 if(parentPointer != nullptr) {
348 // We add pass the components to the parent coupled model
349 for (auto& [componentId, component]: components) {
350 parentPointer->addComponent(component);
351 }
352 // We adapt EIC couplings and add to the parent coupled model
353 leftCouplings(serialEIC, parentPointer->serialEIC);
354 leftCouplings(serialEIC, parentPointer->serialIC);
355 // We add pass the ICs to the parent coupled model
356 parentPointer->serialIC.insert(parentPointer->serialIC.end(), serialIC.begin(), serialIC.end());
357 // We adapt EOC couplings and add to the parent coupled model
358 rightCouplings(serialEOC, parentPointer->serialIC);
359 rightCouplings(serialEOC, parentPointer->serialEOC);
360 }
361 }

368 [[nodiscard]] static MappedCouplings deserializeCouplings(const SerialCouplings& serial) {
369 MappedCouplings map;
370 for(auto& [portFrom, portTo]: serial){
371 addCouplingToMap(map, portFrom, portTo); // this static method checks that there are no duplicates!
372 }
373 return map;
374 }

380 void removeFlattenedCouplings(const std::shared_ptr<Coupled>& child) {
381 for (auto itc = serialEIC.begin(); itc != serialEIC.end(); itc++) {
382 if (child->containsInPort(std::get<1>(*itc))) {
383 serialEIC.erase(itc--);
384 }
385 }
386 for (auto itc = serialIC.begin(); itc != serialIC.end(); itc++) {
387 if (child->containsOutPort(std::get<0>(*itc)) || child->containsInPort(std::get<1>(*itc))) {
388 serialIC.erase(itc--);
389 }
390 }
391 for (auto itc = serialEOC.begin(); itc != serialEOC.end(); itc++) {
392 if (child->containsOutPort(std::get<0>(*itc))) {
393 serialEOC.erase(itc--);
394 }
395 }
396 }

403 void leftCouplings(const SerialCouplings& componentEIC, SerialCouplings& parentCouplings) {
404 // First, we identify component's input ports affected by parent couplings
405 MappedCouplings leftBridge;
406 for (const auto& [portFrom, portTo]: parentCouplings) {
407 if (containsInPort(portTo)) { // If destination port is an input port, we add it to the bridge
408 addCouplingToMap(leftBridge, portFrom, portTo);
409 }
410 }
411 // Then, we detect affected couplings and add them accordingly to the parent
412 for (auto& [left, portTo]: componentEIC) {
413 if (leftBridge.find(left) != leftBridge.end()) {
414 for (auto& portFrom: leftBridge.at(left)) {
415 parentCouplings.emplace_back(portFrom, portTo);
416 }
417 }
418 }
419 }
420
426 void rightCouplings(const SerialCouplings& componentEOC, SerialCouplings& parentCouplings) {
427 // First, we identify component's output ports affected by parent couplings
428 MappedCouplings rightBridge;
429 for (const auto& [portFrom, portTo]: parentCouplings) {
430 if (containsOutPort(portFrom)) { // If origin port is an output port, we add it to the bridge
431 addCouplingToMap(rightBridge, portTo, portFrom); // IMPORTANT: topology of right bridges is inverse
432 }
433 }
434 // Then, we detect affected couplings and add them accordingly to the parent
435 for (auto& [portFrom, right]: componentEOC) {
436 if (rightBridge.find(right) != rightBridge.end()) {
437 for (auto& portTo: rightBridge.at(right)) {
438 parentCouplings.emplace_back(portFrom, portTo);
439 }
440 }
441 }
