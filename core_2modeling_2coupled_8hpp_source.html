149 [[nodiscard]] static bool containsCoupling(const MappedCouplings& couplings, const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
150 if (couplings.find(portTo) == couplings.end()) {
151 return false;
152 }
153 const auto& portsFrom = couplings.at(portTo);
154 return std::find(portsFrom.begin(), portsFrom.end(), portFrom) != portsFrom.end();

164 static void addCouplingToMap(MappedCouplings& coupList, const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
165 auto aux = coupList.find(portTo);
166 if (aux == coupList.end()) {
167 coupList[portTo] = {portFrom};
168 } else {
169 auto& portsFrom = aux->second;
170 if (std::find(portsFrom.begin(), portsFrom.end(), portFrom) != portsFrom.end()) {
171 throw CadmiumModelException("duplicate coupling");
172 }
173 portsFrom.push_back(portFrom);
174 }
175 }

183 void addCoupling(const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
184 if (!portTo->compatible(portFrom)) {
185 throw CadmiumModelException("invalid port type");
186 }
187 if (portFrom->getParent() == nullptr || portTo->getParent() == nullptr) {
188 throw CadmiumModelException("port does not belong to any model");
189 }
190 auto portFromParent = portFrom->getParent();
191 auto portToParent = portTo->getParent();
192 if (containsInPort(portFrom)) {
193 if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
194 addEIC(portFrom, portTo);
195 } else {
196 throw CadmiumModelException("invalid destination port");
197 }
198 } else if (portFromParent->getParent() == this && portFromParent->containsOutPort(portFrom)) {
199 if (containsOutPort(portTo)) {
200 addEOC(portFrom, portTo);
201 } else if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
202 addIC(portFrom, portTo);
203 } else {
204 throw CadmiumModelException("invalid destination port");
205 }
206 } else {
207 throw CadmiumModelException("invalid origin port");
208 }
209 }

315 void flatten(Coupled * parentPointer) {
316 // First, we identify the coupled subcomponents that need to be flattened
317 std::vector<std::shared_ptr<Coupled>> toFlatten;
318 for (auto& component: components) {
319 auto coupled = std::dynamic_pointer_cast<Coupled>(component.second);
320 if (coupled != nullptr) {
321 toFlatten.push_back(coupled);
322 }
323 }
324 // Next, we flatten all the coupled subcomponents
325 for (auto& coupled: toFlatten) {
326 coupled->flatten(this);
327 removeFlattenedCouplings(coupled);
328 components.erase(coupled->getId());
329 }
330 // Finally, we deserialize the resulting couplings. The model is already flat!
331 EIC = deserializeCouplings(serialEIC);
332 IC = deserializeCouplings(serialIC);
333 EOC = deserializeCouplings(serialEOC);
334
335 // If pointer to parent is not null, we propagate the flattening to the corresponding parent coupled model.
336 if(parentPointer != nullptr) {
337 // We add pass the components to the parent coupled model
338 for (auto& [componentId, component]: components) {
339 parentPointer->addComponent(component);
340 }
341 // We adapt EIC couplings and add to the parent coupled model
342 leftCouplings(serialEIC, parentPointer->serialEIC);
343 leftCouplings(serialEIC, parentPointer->serialIC);
344 // We add pass the ICs to the parent coupled model
345 parentPointer->serialIC.insert(parentPointer->serialIC.end(), serialIC.begin(), serialIC.end());
346 // We adapt EOC couplings and add to the parent coupled model
347 rightCouplings(serialEOC, parentPointer->serialIC);
348 rightCouplings(serialEOC, parentPointer->serialEOC);
349 }
350 }

357 [[nodiscard]] static MappedCouplings deserializeCouplings(const SerialCouplings& serial) {
358 MappedCouplings map;
359 for(auto& [portFrom, portTo]: serial){
360 addCouplingToMap(map, portFrom, portTo); // this static method checks that there are no duplicates!
361 }
362 return map;
363 }

369 void removeFlattenedCouplings(const std::shared_ptr<Coupled>& child) {
370 for (auto itc = serialEIC.begin(); itc != serialEIC.end(); itc++) {
371 if (child->containsInPort(std::get<1>(*itc))) {
372 serialEIC.erase(itc--);
373 }
374 }
375 for (auto itc = serialIC.begin(); itc != serialIC.end(); itc++) {
376 if (child->containsOutPort(std::get<0>(*itc)) || child->containsInPort(std::get<1>(*itc))) {
377 serialIC.erase(itc--);
378 }
379 }
380 for (auto itc = serialEOC.begin(); itc != serialEOC.end(); itc++) {
381 if (child->containsOutPort(std::get<0>(*itc))) {
382 serialEOC.erase(itc--);
383 }
384 }
385 }

392 void leftCouplings(const SerialCouplings& componentEIC, SerialCouplings& parentCouplings) {
393 // First, we identify component's input ports affected by parent couplings
394 MappedCouplings leftBridge;
395 for (const auto& [portFrom, portTo]: parentCouplings) {
396 if (containsInPort(portTo)) { // If destination port is an input port, we add it to the bridge
397 addCouplingToMap(leftBridge, portFrom, portTo);
398 }
399 }
400 // Then, we detect affected couplings and add them accordingly to the parent
401 for (auto& [left, portTo]: componentEIC) {
402 if (leftBridge.find(left) != leftBridge.end()) {
403 for (auto& portFrom: leftBridge.at(left)) {
404 parentCouplings.emplace_back(portFrom, portTo);
405 }
406 }
407 }
408 }
409
415 void rightCouplings(const SerialCouplings& componentEOC, SerialCouplings& parentCouplings) {
416 // First, we identify component's output ports affected by parent couplings
417 MappedCouplings rightBridge;
418 for (const auto& [portFrom, portTo]: parentCouplings) {
419 if (containsOutPort(portFrom)) { // If origin port is an output port, we add it to the bridge
420 addCouplingToMap(rightBridge, portTo, portFrom); // IMPORTANT: topology of right bridges is inverse
421 }
422 }
423 // Then, we detect affected couplings and add them accordingly to the parent
424 for (auto& [portFrom, right]: componentEOC) {
425 if (rightBridge.find(right) != rightBridge.end()) {
426 for (auto& portTo: rightBridge.at(right)) {
427 parentCouplings.emplace_back(portFrom, portTo);
428 }
429 }
430 }
