66 [[nodiscard]] bool cellInScenario(const coordinates& cell) const {
67 return cell.size() == shape.size() && std::all_of(cell.begin(), cell.end(), [this, i = 0](int a) mutable {
68 a = a - origin[i];
69 return a >= 0 && a < shape[i++];

78 [[nodiscard]] bool validDistance(const coordinates& distance) const {
79 return distance.size() == shape.size() && std::all_of(distance.begin(), distance.end(), [this, i = 0](int a) mutable {
80 return std::abs(a) < shape[i++];

90 [[nodiscard]] coordinates distanceVector(const coordinates& cellFrom, const coordinates& cellTo) const {
91 if (!cellInScenario(cellFrom) || !cellInScenario(cellTo)) {
92 throw CadmiumModelException("Cell does not belong to scenario");
93 }
94 coordinates distance;
95 for (int i = 0; i < shape.size(); ++i) {
96 auto d = cellTo[i] - cellFrom[i];
97 if (wrapped && abs(d) > shape[i] / 2) {
98 d -= (int) copysign(shape[i], d);
99 }
100 distance.push_back(d);
101 }
102 if (!validDistance(distance)) {
103 throw CadmiumModelException("Invalid distance vector");
104 }
105 return distance;
106 }

114 [[maybe_unused]] [[nodiscard]] coordinates cellTo(const coordinates& cellFrom, const coordinates& distance) const {
115 if (!cellInScenario(cellFrom)) {
116 throw CadmiumModelException("Cell does not belong to scenario");
117 }
118 if (!validDistance(distance)) {
119 throw CadmiumModelException("Invalid distance vector");
120 }
121 coordinates cellTo;
122 for (int i = 0; i < shape.size(); ++i) {
123 auto v = cellFrom[i] + distance[i];
124 if (wrapped) {
125 v = (v + shape[i]) % shape[i];
126 }
127 cellTo.push_back(v);
128 }
129 if (!cellInScenario(cellTo)) {
130 throw CadmiumModelException("Cell does not belong to scenario");
131 }
132 return cellTo;
133 }

141 [[maybe_unused]] [[nodiscard]] coordinates cellFrom(const coordinates& distance, const coordinates& cellTo) const {
142 if (!validDistance(distance)) {
143 throw CadmiumModelException("Invalid distance vector");
144 }
145 if (!cellInScenario(cellTo)) {
146 throw CadmiumModelException("Cell does not belong to scenario");
147 }
148 coordinates cellFrom;
149 for (int i = 0; i < shape.size(); ++i) {
150 auto v = cellTo[i] - distance[i];
151 if (wrapped) {
152 v = (v + shape[i]) % shape[i];
153 }
154 cellFrom.push_back(v);
155 }
156 if (!cellInScenario(cellFrom)) {
157 throw CadmiumModelException("Cell does not belong to scenario");
158 }
159 return cellFrom;

199 [[nodiscard]] static double minkowskiDistance(int p, const coordinates& distance) {
200 if (p < 1) {
201 throw CadmiumModelException("p must be greater than 0");
202 }
203 auto x = std::accumulate(distance.begin(), distance.end(), 0., [p](double sum, int v) { return sum + std::pow(std::abs(v), p); });
204 return std::pow(x, 1.0 / p);
205 }

296 void next(int d) {

303 next(d + 1);
