55 [[nodiscard]] bool cellInScenario(const coordinates& cell) const {
56 return cell.size() == shape.size() && std::all_of(cell.begin(), cell.end(), [this, i = 0](int a) mutable {
57 a = a - origin[i];
58 return a >= 0 && a < shape[i++];
59 });

67 [[nodiscard]] bool validDistance(const coordinates& distance) const {
68 return distance.size() == shape.size() && std::all_of(distance.begin(), distance.end(), [this, i = 0](int a) mutable {
69 return std::abs(a) < shape[i++];
70 });

79 [[nodiscard]] coordinates distanceVector(const coordinates& cellFrom, const coordinates& cellTo) const {
80 if (!cellInScenario(cellFrom) || !cellInScenario(cellTo)) {
81 throw CadmiumModelException("Cell does not belong to scenario");
82 }
83 coordinates distance;
84 for (int i = 0; i < shape.size(); ++i) {
85 auto d = cellTo[i] - cellFrom[i];
86 if (wrapped && abs(d) > shape[i] / 2) {
87 d -= (int) copysign(shape[i], d);
88 }
89 distance.push_back(d);
90 }
91 if (!validDistance(distance)) {
92 throw CadmiumModelException("Invalid distance vector");
93 }
94 return distance;
95 }

103 [[maybe_unused]] [[nodiscard]] coordinates cellTo(const coordinates& cellFrom, const coordinates& distance) const {
104 if (!cellInScenario(cellFrom)) {
105 throw CadmiumModelException("Cell does not belong to scenario");
106 }
107 if (!validDistance(distance)) {
108 throw CadmiumModelException("Invalid distance vector");
109 }
110 coordinates cellTo;
111 for (int i = 0; i < shape.size(); ++i) {
112 auto v = cellFrom[i] + distance[i];
113 if (wrapped) {
114 v = (v + shape[i]) % shape[i];
115 }
116 cellTo.push_back(v);
117 }
118 if (!cellInScenario(cellTo)) {
119 throw CadmiumModelException("Cell does not belong to scenario");
120 }
121 return cellTo;
122 }

130 [[maybe_unused]] [[nodiscard]] coordinates cellFrom(const coordinates& distance, const coordinates& cellTo) const {
131 if (!validDistance(distance)) {
132 throw CadmiumModelException("Invalid distance vector");

134 if (!cellInScenario(cellTo)) {
135 throw CadmiumModelException("Cell does not belong to scenario");

137 coordinates cellFrom;
138 for (int i = 0; i < shape.size(); ++i) {
139 auto v = cellTo[i] - distance[i];
140 if (wrapped) {

142 }
143 cellFrom.push_back(v);

188 [[nodiscard]] static double minkowskiDistance(int p, const coordinates& distance) {
189 if (p < 1) {
190 throw CadmiumModelException("p must be greater than 0");
191 }
192 auto x = std::accumulate(distance.begin(), distance.end(), 0., [p](double sum, int v) { return sum + std::pow(std::abs(v), p); });
193 return std::pow(x, 1.0 / p);
194 }

201 [[nodiscard]] static inline int manhattanDistance(const coordinates& distance) {
202 return std::accumulate(distance.begin(), distance.end(), 0, [](int sum, int v) { return sum + std::abs(v); });
