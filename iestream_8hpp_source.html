48 std::pair<double, std::optional<MSG>> nextTimedInput() {
49 // Default return values: infinity and none
50 double sigma = std::numeric_limits<double>::infinity();
51 std::optional<MSG> contents = std::optional<MSG>();
52 if (file.is_open() && !file.eof()) {
53 file >> sigma; // read time of next message
54 MSG value;
55 file >> value; // read values to go into message
56 contents.emplace(value);
57 }
58 return std::make_pair(sigma,contents);

119 void internalTransition(IEStreamState<MSG>& state) const override {
120 state.clock += state.sigma;
121 while(true) { // loop to ignore outdated events
122 auto [nextTime, nextEvent] = state.parser.nextTimedInput();
123 if (nextTime < state.clock) {
124 std::cerr << "Outdated event in input file. Scheduled time: " << nextTime << " Current time: " << state.clock << ". " << nextEvent.value() << std::endl;
125 } else {
126 state.sigma = nextTime - state.clock;
127 state.lastInputRead = nextEvent;
128 break;
129 }
130 };
131 }
