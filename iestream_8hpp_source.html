60  // Default return values: infinity and none

62  std::optional<MSG> contents = std::optional<MSG>();

130  void internalTransition(IEStreamState<MSG>& state) const override {
131  state.clock += state.sigma;
132  while(true) { // loop to ignore outdated events
133  auto [nextTime, nextEvent] = state.parser.nextTimedInput();
134  if (nextTime < state.clock) {
135  std::cerr << "Outdated event in input file. Scheduled time: " << nextTime << " Current time: " << state.clock << ". " << nextEvent.value() << std::endl;
136  } else {
137  state.sigma = nextTime - state.clock;
138  state.lastInputRead = nextEvent;
139  break;
