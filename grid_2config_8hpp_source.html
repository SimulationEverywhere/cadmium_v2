67  void processNeighborhood() {
68  for (const nlohmann::json& rawNeighbors: rawNeighborhood) {
69  auto type = rawNeighbors.at("type").get<std::string>();
70  auto vicinity = (rawNeighbors.contains("vicinity")) ? rawNeighbors["vicinity"].get<V>() : V();
71  if (type == "absolute") {
72  auto neighbors = rawNeighbors.at("neighbors").get<std::vector<coordinates>>();
73  for (const auto& neighbor: neighbors) {
74  if (scenario->cellInScenario(neighbor)) {
75  absolute[neighbor] = NeighborData<S, V>(vicinity);
76  } else {
77  throw CadmiumModelException("Invalid absolute neighbor");
78  }
79  }
80  } else {
81  std::vector<coordinates> distances;
82  if (type == "relative") {
83  distances = rawNeighbors.at("neighbors").get<std::vector<coordinates>>();
84  }
85  else if (type == "moore") {
86  auto range = (rawNeighbors.contains("range")) ? rawNeighbors["range"].get<int>() : 1;
87  distances = scenario->mooreNeighborhood(range);
88  }
89  else if (type == "chebyshev" || type == "chessboard") {
90  auto range = (rawNeighbors.contains("range")) ? (int)rawNeighbors["range"].get<double>() : 1;
91  distances = scenario->mooreNeighborhood(std::floor(range));
92  }
93  else if (type == "von_neumann") {
94  auto range = (rawNeighbors.contains("range")) ? rawNeighbors["range"].get<int>() : 1;
95  distances = scenario->vonNeumannNeighborhood(range);
96  }
97  else if (type == "manhattan" || type == "taxicab" || type == "snake") {
98  auto range = (rawNeighbors.contains("range")) ? (int)rawNeighbors["range"].get<double>() : 1;
99  distances = scenario->vonNeumannNeighborhood(range);
100  }
101  else if (type == "minkowski" || type == "euclidean") {
102  auto p = (type == "euclidean") ? 2 : rawNeighbors.at("p").get<int>();
103  auto range = (rawNeighbors.contains("range")) ? rawNeighbors["range"].get<double>() : 1.;
104  distances = scenario->minkowskiNeighborhood(p, range);
105  }
106  else {
107  throw CadmiumModelException("Unknown neighborhood type");
108  }
109  for (const auto& distance: distances) {
110  if (scenario->validDistance(distance)) {
111  relative[distance] = NeighborData<S, V>(vicinity);
112  } else {
113  throw CadmiumModelException("Invalid distance vector");
114  }
115  }
116  }
117  }
118  }

125  std::unordered_map<coordinates, NeighborData<S, V>> buildNeighborhood(const coordinates& cellId) const override {
126  std::unordered_map<coordinates, NeighborData<S, V>> neighborhood;
127  for (const auto& [distance, vicinity]: relative) {
128  try {
129  neighborhood[scenario->cellTo(cellId, distance)] = vicinity;
130  } catch (CadmiumModelException&) {
131  continue;
132  }
133  }
134  for (const auto& [neighbor, vicinity]: absolute) {
135  neighborhood[neighbor] = vicinity;
136  }
137  return neighborhood;
138  }
